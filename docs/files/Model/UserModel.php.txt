namespace App\Model;


use App\Core\Database;

class UserModel
{
    /**
     * @var Database
     */
    private $db;

    public function __construct(Database $database)
    {
        $this->db = $database;
    }

    public function findAll() {
        return $this->db->query("SELECT * FROM users")->fetchAll();
    }

}
